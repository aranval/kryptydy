Niżej przykład dialogu gotowy do wklejenia. Dialogi muszą mieć unikalną nazwe.

Dialogues.nazwaDialogu = {
    -- Warunki do spełnienia aby dialog sie wyświetlił
    -- Można użyć:
    -- < - mniejsze, > - większe, == - równe
    -- and - i , or - lub
    -- Dostępne gameEvents z pliku "GmameEvents.txt"
    -- Dostepne Inventory:
    -- inventory:getNumberOfItem(itemID) -- zwraca wartość itemu, itemID to string musi być w "".
    -- itemID z pliku Assets/Items.txt
    function()
        return gameEvents.event1 == 0 and inventory:getNumberOfItem("testItem") == 1
    end,
    -- Dialog. other - mówi npc, self - mówi gracz
    {"other", "Mówi npc"},
    {"self", "Mówi player"}
    -- Funkcja wykonuje się po dialogu
    -- Można zmienić statusu eventów
    -- Dostepne Inventory:
    -- inventory:addItem(itemID) - dodaje item
    -- inventory:removeItem(itemID) - dodaje item
    function() 
        gameEvents.event1 = 1
        inventory:removeItem("testItem")
    end
}
